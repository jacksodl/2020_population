countries = population['Country'].head(15)

fig, ax1 = plt.subplots(figsize=(15, 10))
# plt.hexbin(countries, , cmap=plt.cm.gnuplot2_r) # Commas in Y axis

ax2 = ax1.twinx()
ax1.bar(countries, pop, color='b')
ax2.plot(countries, avg_pop, 'r-', linewidth=5)

ax1.set_xlabel('Country', color='g', size = 15, weight='bold')
ax1.set_ylabel('Population', color='b', size = 15, weight='bold')
ax1.ticklabel_format(style='plain', axis='y')
ax2.set_ylabel('Average Population', color='r', size = 15)
ax1.set_xticklabels(countries, rotation = 55, weight='bold')
ax1.set_yticklabels(pop, weight='bold')
ax1.yaxis.set_major_formatter(plt.FuncFormatter(lambda y,  p: format(int(y), ',')))
ax2.yaxis.set_major_formatter(plt.FuncFormatter(lambda y,  p: format(int(y), ',')))
plt.title('2020 Population', size=30, weight='bold')
